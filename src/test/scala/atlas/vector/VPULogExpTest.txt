package atlas.vector

import chisel3._
import chisel3.simulator.EphemeralSimulator._
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import atlas.common.VPUParams
import org.scalatest.Outcome

class VPULogExpTest extends AnyFlatSpec with Matchers {

  // --- BF16 HEX Numbers ---
  val BF16_0      = 0x0000   // +0.0
  val BF16_0_5    = 0x3F00   //  0.5
  val BF16_1      = 0x3F80   //  1.0
  val BF16_1_5    = 0x3FC0   //  1.5
  val BF16_2      = 0x4000   //  2.0
  val BF16_3      = 0x4040   //  3.0
  val BF16_N1     = 0xBF80   // -1.0
  val BF16_N0_5   = 0xBF00   // -0.5
  val BF16_MIN_NORMAL = 0x0080 // smallest positive normal

  val p = VPUParams()
  behavior of s"VectorEngine (BF16) log/exp lanes=${p.numLanes}"

  // HElPER FUNCTIONS
  
  def initializeInterface(dut: VectorEngine, drainCycles: Int = 4): Unit = {
    dut.io.in.valid.poke(false.B)
    dut.io.out.ready.poke(true.B)
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(0.U)
      dut.io.in.bits.vecDataB(i).poke(0.U)
    }
    if (drainCycles > 0) dut.clock.step(drainCycles)
  }
  
  // Float32 -> BF16- format converters so we can check for precision!! sine the HEX won't actually match
  // 16 bit BF16 litraly becomes top 16 bits of a fp32, and converts it to a float
  def bf16ToFloat(bf16: Int): Float = {
    val bits32 = (bf16 & 0xFFFF) << 16
    java.lang.Float.intBitsToFloat(bits32)
  }

  // makes checking our precision customizable
  def expectNear(actual: Float, expected: Float, absTol: Float, relTol: Float, name: String): Unit = {
    val absErr = Math.abs(actual - expected).toFloat
    val relErr = (absErr / Math.max(Math.abs(expected), 1e-12f)).toFloat
    assert(
      absErr <= absTol || relErr <= relTol,
      f"$name failed: actual=$actual%.8f expected=$expected%.8f absErr=$absErr%.8f relErr=$relErr%.8f"
    )
  }

    def runOpAndCheck(dut: VectorEngine, op: VPUOp.Type, a: Int, b: Int, expectedFloat: Float, absTol: Float, 
    relTol: Float, opName: String ): Unit = {
    dut.io.in.valid.poke(true.B)
    dut.io.in.bits.instType.poke(op)
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(a.U)
      dut.io.in.bits.vecDataB(i).poke(b.U)
    }

    var cycles = 0
    while (!dut.io.in.ready.peek().litToBoolean && cycles < 50) {
      dut.clock.step(1); cycles += 1
    }
    dut.io.in.ready.expect(true.B, s"Timed out waiting for in.ready on $opName")

    dut.clock.step(1)
    dut.io.in.valid.poke(false.B)

    cycles = 0
    while (!dut.io.out.valid.peek().litToBoolean && cycles < 300) {
      dut.clock.step(1); cycles += 1
    }
    dut.io.out.valid.expect(true.B, s"Timed out waiting for out.valid on $opName")

    println(s"--- Test: $opName ---")
    println(f"Input A : 0x$a%04X")
    println(f"Input B : 0x$b%04X")
    println(f"Expected (float): $expectedFloat%.8f")

    for (i <- 0 until dut.p.numLanes) {
      val outBits = dut.io.out.bits.vectorOutputData(i).peek().litValue.toInt & 0xFFFF
      val outF = bf16ToFloat(outBits)
      if (i == 0) println(f"Output  : 0x$outBits%04X  ($outF%.8f)")
      expectNear(outF, expectedFloat, absTol, relTol, s"$opName lane $i")
    }
    println("---------------------\n")

    dut.clock.step(1)
  }


//EXP 
it should "EXP: exp(0)=1 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.exp, BF16_0, BF16_0, 1.0f, 0.01f, 0.01f, "exp(0)")
    }
  }

  it should "EXP: exp(1)≈2.718 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.exp, BF16_1, BF16_0, Math.E.toFloat, 0.15f, 0.07f, "exp(1)")
    }
  }

  it should "EXP: exp(-1)≈0.3679 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.exp, BF16_N1, BF16_0, (1.0 / Math.E).toFloat, 0.05f, 0.12f, "exp(-1)")
    }
  }

  // ALL LOGS
  it should "LOG: log(1)=0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.log, BF16_1, BF16_0, 0.0f, 0.02f, 0.0f, "log(1)")
    }
  }

  it should "LOG: log(2)=1 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.log, BF16_2, BF16_0, (Math.log(2.0)/Math.log(2.0)).toFloat, 0.08f, 0.12f, "log(2)")
    }
  }

  it should "LOG: log(0.5)=-1 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.log, BF16_0_5, BF16_0, (Math.log(0.5)/Math.log(2.0)).toFloat, 0.08f, 0.12f, "log(0.5)")
    }
  }

  it should "LOG: log(1.5)≈0.584962 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.log, BF16_1_5, BF16_0, (Math.log(1.5)/Math.log(2.0)).toFloat, 0.08f, 0.12f, "log(0.5)")
    }
  }
  
  it should "LOG: log(6.2)≈2.63226 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.log, 0x40c6, BF16_0, (Math.log(6.2)/Math.log(2.0)).toFloat, 0.08f, 0.12f, "log(0.5)")
    }
  }
}
