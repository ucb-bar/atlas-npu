package atlas.vector

import chisel3._
import chisel3.simulator.EphemeralSimulator._
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import atlas.common.VPUParams
import sp26FPUnits.hardfloat._
import org.scalatest.Outcome


class TanhLUTSpec extends AnyFlatSpec with Matchers {
  //----------- CI/CD INCLUDE --------------
  override def withFixture(test: NoArgTest): Outcome = {
    val outcome = super.withFixture(test)
    if (outcome.isFailed) {
      println("TanhLUTSpec=FAILED")
    }
    outcome
  }
  //----------- CI/CD INCLUDE --------------
  val ports = 1
  val addrBits = 8
  val m = 4
  val n = 12
  val entries = 1 << addrBits

  // Golden Model: Replicates the EXACT math in your RTL
  def goldenTanh(i: Int): BigInt = {
    val max = math.Pi / 2
    val min = 0.0
    val r = min + (i + 1).toDouble * (max - min) / entries
    val v = math.tanh(r)
    BigInt(math.round(v * (1 << n)))
  }

  behavior of "TanhLUT"

  it should "verify the full LUT contents and 1-cycle latency" in {
    simulate(new TanhLUT(ports, addrBits, m, n)) { dut =>
      // Sweep every address
      for (i <- 0 until entries) {
        dut.io.ren(0).poke(true.B)
        dut.io.raddr(0).poke(i.U)
        
        dut.clock.step(1) // Wait for register update
        
        val expectedY0 = goldenTanh(i)
        val nextIdx = if (i == entries - 1) i else i + 1
        val expectedY1 = goldenTanh(nextIdx)

        dut.io.rdata(0)(0).expect(expectedY0.U)
        dut.io.rdata(1)(0).expect(expectedY1.U)
      }
    }
  }

  it should "retain stale data when ren is low" in {
    simulate(new TanhLUT(ports, addrBits, m, n)) { dut =>
      // 1. Load a value
      dut.io.ren(0).poke(true.B)
      dut.io.raddr(0).poke(50.U)
      dut.clock.step(1)
      val firstValue = dut.io.rdata(0)(0).peek().litValue // Updated peek syntax

      // 2. Disable enable and change address
      dut.io.ren(0).poke(false.B)
      dut.io.raddr(0).poke(100.U)
      dut.clock.step(1)

      // 3. Output should NOT have changed to Value 100
      dut.io.rdata(0)(0).expect(firstValue.U)
    }
  }
}
