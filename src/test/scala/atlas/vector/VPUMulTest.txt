package atlas.vector

import chisel3._
import chisel3.simulator.EphemeralSimulator._
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import atlas.common.VPUParams
import sp26FPUnits.hardfloat._
import org.scalatest.Outcome

class VPUMulTest extends AnyFlatSpec with Matchers {

  //--- BF16 HEX Numbers ---
  // Integers
  val BF16_N2    = 0xC000   // -2.0
  val BF16_N1 = 0xBF80   // -1.0
  val BF16_N0 = 0x8000   // -0.0
  val BF16_0  = 0x0000   // +0.0
  val BF16_1  = 0x3F80   //  1.0
  val BF16_2  = 0x4000   //  2.0
  val BF16_3  = 0x4040   //  3.0
  val BF16_4  = 0x4080   //  4.0
  val BF16_5  = 0x40A0   //  5.0

  // Fractional numbers
  val BF16_0_5   = 0x3F00   //  0.5
  val BF16_0_25  = 0x3E80   //  0.25
  val BF16_1_25  = 0x3FA0   //  1.25
  val BF16_1_5   = 0x3FC0   //  1.5
  val BF16_N1_5  = 0xBFC0   // -1.5

  // Approximate decimal- our roudning 
  val BF16_1_1   = 0x3F8D   // ≈ 1.1015625 (rounded representation of 1.1)
  val BF16_1_PLUS_EPS = 0x3F81   // 1.0078125 (smallest number above 1.0)

  // Extremes
  val BF16_MAX        = 0x7F7F   // largest finite value
  val BF16_MIN_NORMAL = 0x0080   // smallest positive normal (2^-126)
  val BF16_MIN_DENORM    = 0x0001   // smallest positive denormalized # (2^-133)

  val p = VPUParams() 
  behavior of s"VectorEngine (BF16) mul/div/sqrt lanes=${p.numLanes}"

  // -------------------------------------------------------------------------
  // Helper Functions
  // -------------------------------------------------------------------------
  def initializeInterface(dut: VectorEngine, drainCycles: Int = 4): Unit = {
    dut.io.in.valid.poke(false.B)
    dut.io.out.ready.poke(true.B) // Always ready to accept output in this test
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(0.U)
      dut.io.in.bits.vecDataB(i).poke(0.U)
    }
    if (drainCycles > 0) dut.clock.step(drainCycles)
  }

  // Unified helper to send data, wait for FSM to finish, check, and print
  def runOpAndCheck(dut: VectorEngine, op: VPUOp.Type, a: Int, b: Int, expected: Int, opName: String): Unit = {
    // 1. Setup and Fire Input
    dut.io.in.valid.poke(true.B)
    dut.io.in.bits.instType.poke(op)
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(a.U)
      dut.io.in.bits.vecDataB(i).poke(b.U)
    }

    // Wait for in.ready to be true
    var cycles = 0
    while (!dut.io.in.ready.peek().litToBoolean && cycles < 20) {
      dut.clock.step(1)
      cycles += 1
    }
    dut.io.in.ready.expect(true.B, s"Timed out waiting for in.ready on $opName")
    
    dut.clock.step(1) // Advance clock to complete the input handshake (FSM goes to BUSY)
    dut.io.in.valid.poke(false.B) // Drop valid so we don't double-fire

    // 2. Wait for Output Valid (FSM finishes calculating)
    cycles = 0
    while (!dut.io.out.valid.peek().litToBoolean && cycles < 100) {
      dut.clock.step(1)
      cycles += 1
    }
    dut.io.out.valid.expect(true.B, s"Timed out waiting for out.valid on $opName")

    // 3. Print and Check Values
    println(s"--- Test: $opName ---")
    println(f"Input A : 0x$a%04X")
    println(f"Input B : 0x$b%04X")
    println(f"Expected: 0x$expected%04X")

    for (i <- 0 until dut.p.numLanes) {
      val actualVal = dut.io.out.bits.vectorOutputData(i).peek().litValue.toInt
      if (i == 0) {
        // Only print lane 0 to keep the console clean
        println(f"Output  : 0x$actualVal%04X") 
      }
      dut.io.out.bits.vectorOutputData(i).expect(expected.U, s"Mismatch on lane $i")
    }
    println("---------------------\n")

    // 4. Complete Output Handshake (FSM returns to IDLE)
    // out.ready is already poked true in initializeInterface
    dut.clock.step(1) 
  }

  // -------------------------------------------------------------------------
  // Test Cases
  // -------------------------------------------------------------------------
  it should "MUL: 1.0 * 1.0 = 1.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_1, BF16_1, BF16_1, "1.0 * 1.0")
    }
  }

  it should "MUL: 2.0 * 2.0 = 4.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_2, BF16_2, BF16_4, "2.0 * 2.0")
    }
  }

  it should "MUL: 1.0 * 2.0 = 2.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_1, BF16_2, BF16_2, "1.0 * 2.0")
    }
  }

  it should "MUL: 0.0 * 1.0 = 0.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_0, BF16_1, BF16_0, "0.0 * 1.0")
    }
  }

  it should "MUL: (-0.0) * 1.0 = (-0.0) (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_N0, BF16_1, BF16_N0, "-0.0 * 1.0")
    }
  }

  it should "MUL: (-0.0) * 0.0 = (-0.0) (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_N0, BF16_0, BF16_N0, "-0.0 * 0.0")
    }
  }

  it should "MUL: (-1.0) * 1.0 = -1.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_N1, BF16_1, BF16_N1, "-1.0 * 1.0")
    }
  }

  // fractional
  it should "MUL: 1.5 * 2.0 = 3.0  (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_1_5, BF16_2, BF16_3, "1.5 * 2.0")
    }
  }

  it should "MUL: 0.5 * 0.5 = 0.25  (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_0_5, BF16_0_5, BF16_0_25, "0.5 * 0.5")
    }
  }

  it should "MUL: -1.5 * 2.0 = -3.0  (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_N1_5, BF16_2, 0xC040, "-1.5 * 2.0")
    }
  }

  it should "MUL: -1.5 * -2.0 = 3.0  (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_N1_5, BF16_N2, BF16_3, "-1.5 * -2.0")
    }
  }

  // rounding
  it should "MUL: 1.1 * 1.1 (rounding case)  (all lanes)" in {
  simulate(new VectorEngine(p)) { dut =>
    initializeInterface(dut)
    // Expected ≈ 1.21 → BF16 rounded ≈ 0x3F9B
    runOpAndCheck(dut, VPUOp.mul, BF16_1_1, BF16_1_1, 0x3F9B, "1.1 * 1.1")
    }
  }

  // Underflow
  it should "MUL: MIN_NORMAL * 0.5 (underflow toward denoramlized) (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, BF16_MIN_NORMAL, BF16_0_5, 0x0040, "underflow case")
    }
  }

  /// denormalized nums
  it should "MUL: MIN_DENORM * 2 = denormalized (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, 0x0001, 0x4000, 0x0002, "min_denorm * 2")
    }
  }

  it should "MUL: MIN_DENORM * 4 = next-next denorm (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, 0x0001, 0x4080, 0x0004, "min_denorm * 4")
    }
  }

  it should "MUL: MIN_DENORM * MIN_DENORM = 0 (underflow to zero) (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, 0x0001, 0x0001, 0x0000, "tiny * tiny")
    }
  }

  it should "MUL: MIN_NORMAL * 0.5 becomes denroamlized (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, 0x0080, 0x3F00, 0x0040, "min_normal * 0.5")
    }
  }

  it should "MUL: negative MIN_DENORM * 1 keeps sign (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.mul, 0x8001, 0x3F80, 0x8001, "neg_min_denorm * 1")
    }
  }
}
