package atlas.vector

import chisel3._
import chisel3.simulator.EphemeralSimulator._
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import atlas.common.VPUParams
import org.scalatest.Outcome

class VPUSinCosTanhTest extends AnyFlatSpec with Matchers {

    override def withFixture(test: NoArgTest): Outcome = {
        val outcome = super.withFixture(test)
        if (outcome.isFailed) println("VPUSinCosTanhTest=FAILED")
        outcome
    }

    ///--- BF16 HEX Numbers ---
    val BF16_N2    = 0xC000   // -2.0
    val BF16_N1   = 0xBF80  // - 1 
    val BF16_N0_5 = 0xBF00  // -0.5
    val BF16_0    = 0x0000  // 0
    val BF16_0_5  = 0x3F00  // 0.5
    val BF16_1    = 0x3F80  // 1
    val BF16_2    = 0x4000  // 2
    val BF16_PI_2 = 0x3FC9   // ~1.5703125  (≈ pi/2)
    val BF16_PI   = 0x4049   // ~3.140625   (≈ pi)

    val p = VPUParams()
    behavior of s"VectorEngine (BF16) sin/cos/tanh lanes=${p.numLanes}"

    def initializeInterface(dut: VectorEngine, drainCycles: Int = 4): Unit = {
        dut.io.in.valid.poke(false.B)
        dut.io.out.ready.poke(true.B)
        for (i <- 0 until dut.p.numLanes) {
        dut.io.in.bits.vecDataA(i).poke(0.U)
        dut.io.in.bits.vecDataB(i).poke(0.U)
        }
        if (drainCycles > 0) dut.clock.step(drainCycles)
    }

    def bf16ToFloat(bf16: Int): Float = {
        val bits32 = (bf16 & 0xFFFF) << 16
        java.lang.Float.intBitsToFloat(bits32)
    }

    def expectNear(actual: Float, expected: Float, absTol: Float, relTol: Float, name: String): Unit = {
        val absErr = Math.abs(actual - expected).toFloat
        val relErr = (absErr / Math.max(Math.abs(expected), 1e-12f)).toFloat
        assert(
        absErr <= absTol || relErr <= relTol,
        f"$name failed: actual=$actual%.8f expected=$expected%.8f absErr=$absErr%.8f relErr=$relErr%.8f"
        )
    }

    def runOpAndCheck( dut: VectorEngine, op: VPUOp.Type, a: Int, b: Int, expectedFloat: Float, 
                        absTol: Float, relTol: Float, opName: String): Unit = {
        dut.io.in.valid.poke(true.B)
        dut.io.in.bits.instType.poke(op)
        for (i <- 0 until dut.p.numLanes) {
        dut.io.in.bits.vecDataA(i).poke(a.U)
        dut.io.in.bits.vecDataB(i).poke(b.U)
        }

        var cycles = 0
        while (!dut.io.in.ready.peek().litToBoolean && cycles < 50) {
        dut.clock.step(1); cycles += 1
        }
        dut.io.in.ready.expect(true.B, s"Timed out waiting for in.ready on $opName")

        dut.clock.step(1)
        dut.io.in.valid.poke(false.B)

        cycles = 0
        while (!dut.io.out.valid.peek().litToBoolean && cycles < 300) {
        dut.clock.step(1); cycles += 1
        }
        dut.io.out.valid.expect(true.B, s"Timed out waiting for out.valid on $opName")

        println(s"--- Test: $opName ---")
        println(f"Input A : 0x$a%04X")
        println(f"Input B : 0x$b%04X")
        println(f"Expected (float): $expectedFloat%.8f")

        for (i <- 0 until dut.p.numLanes) {
        val outBits = dut.io.out.bits.vectorOutputData(i).peek().litValue.toInt & 0xFFFF
        val outF = bf16ToFloat(outBits)
        if (i == 0) println(f"Output  : 0x$outBits%04X  ($outF%.8f)")
        expectNear(outF, expectedFloat, absTol, relTol, s"$opName lane $i")
        }
        println("---------------------\n")

        dut.clock.step(1)
    }


    // Basic Tests
    it should "SIN: sin(0)=0 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.sin, BF16_0, BF16_0, 0.0f, 0.02f, 0.0f, "sin(0)")
        }
    }

    it should "SIN: sin(0.5)≈0.4794 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.sin, BF16_0_5, BF16_0, Math.sin(0.5).toFloat, 0.06f, 0.12f, "sin(0.5)")
        }
    }

    it should "COS: cos(0)=1 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.cos, BF16_0, BF16_0, 1.0f, 0.03f, 0.03f, "cos(0)")
        }
    }

    it should "COS: cos(0.5)≈0.8776 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.cos, BF16_0_5, BF16_0, Math.cos(0.5).toFloat, 0.06f, 0.10f, "cos(0.5)")
        }
    }

    it should "TANH: tanh(0)=0 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.tanh, BF16_0, BF16_0, 0.0f, 0.02f, 0.0f, "tanh(0)")
        }
    }

    it should "TANH: tanh(1)≈0.7616 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.tanh, BF16_1, BF16_0, Math.tanh(1.0).toFloat, 0.06f, 0.10f, "tanh(1)")
        }
    }

    it should "TANH: tanh(-1)≈-0.7616 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.tanh, BF16_N1, BF16_0, Math.tanh(-1.0).toFloat, 0.06f, 0.10f, "tanh(-1)")
        }
    }

    // More extensive Tests
    it should "SIN: sin(-0.5)≈-sin(0.5) (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.sin, BF16_N0_5, BF16_0, (-Math.sin(0.5)).toFloat, 0.06f, 0.12f, "sin(-0.5)")
        }
    }

    it should "COS: cos(-0.5)≈cos(0.5) (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.cos, BF16_N0_5, BF16_0, Math.cos(0.5).toFloat, 0.06f, 0.10f, "cos(-0.5)")
        }
    }

    it should "SIN: sin(pi/2)≈1 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.sin, BF16_PI_2, BF16_0, 1.0f, 0.08f, 0.08f, "sin(pi/2)")
        }  
    }

    it should "COS: cos(pi)≈-1 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.cos, BF16_PI, BF16_0, (-1.0f), 0.10f, 0.10f, "cos(pi)")
        }
    }

    it should "TANH: tanh(2)≈0.9640 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.tanh, BF16_2, BF16_0, Math.tanh(2.0).toFloat, 0.05f, 0.08f, "tanh(2)")
        }
    }

    it should "TANH: tanh(-2)≈-0.9640 (all lanes)" in {
        simulate(new VectorEngine(p)) { dut =>
        initializeInterface(dut)
        runOpAndCheck(dut, VPUOp.tanh, BF16_N2, BF16_0, Math.tanh(-2.0).toFloat, 0.05f, 0.08f, "tanh(-2)")
        }
    }
}
