package atlas.vector

import chisel3._
import chisel3.simulator.EphemeralSimulator._
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import atlas.common.VPUParams
import sp26FPUnits.hardfloat._

class VPUAddSubTest extends AnyFlatSpec with Matchers {

  // -------------------------------------------------------------------------
  // BF16 bit patterns
  // -------------------------------------------------------------------------
  val BF16_N2 = 0xC000   // -2.0
  val BF16_N1 = 0xBF80   // -1.0
  val BF16_N0 = 0x8000   // -0.0
  val BF16_0  = 0x0000   // +0.0
  val BF16_1  = 0x3F80   //  1.0
  val BF16_2  = 0x4000   //  2.0
  val BF16_3  = 0x4040   //  3.0
  val BF16_4  = 0x4080   //  4.0

  // Fractions (exact in binary)
  val BF16_0_5  = 0x3F00 //  0.5
  val BF16_1_5  = 0x3FC0 //  1.5
  val BF16_2_5  = 0x4020 //  2.5
  val BF16_N0_5 = 0xBF00 // -0.5

  // Rounding boundary helper
  val BF16_1_PLUS_EPS = 0x3F81   // 1.0078125 (next representable above 1.0)
  val BF16_EPS        = 0x3C00   // 0.0078125 = 2^-7 (ULP at 1.0)

  val p = VPUParams()
  behavior of s"VectorEngine (BF16) add/sub lanes=${p.numLanes}"

  // -------------------------------------------------------------------------
  // Helper Functions (same style as VPUMulTest)
  // -------------------------------------------------------------------------
  def initializeInterface(dut: VectorEngine, drainCycles: Int = 4): Unit = {
    dut.io.in.valid.poke(false.B)
    dut.io.out.ready.poke(true.B)
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(0.U)
      dut.io.in.bits.vecDataB(i).poke(0.U)
    }
    if (drainCycles > 0) dut.clock.step(drainCycles)
  }

  def runOpAndCheck(dut: VectorEngine, op: VPUOp.Type, a: Int, b: Int, expected: Int, opName: String): Unit = {
    dut.io.in.valid.poke(true.B)
    dut.io.in.bits.instType.poke(op)
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(a.U)
      dut.io.in.bits.vecDataB(i).poke(b.U)
    }

    var cycles = 0
    while (!dut.io.in.ready.peek().litToBoolean && cycles < 20) {
      dut.clock.step(1); cycles += 1
    }
    dut.io.in.ready.expect(true.B, s"Timed out waiting for in.ready on $opName")

    dut.clock.step(1)
    dut.io.in.valid.poke(false.B)

    cycles = 0
    while (!dut.io.out.valid.peek().litToBoolean && cycles < 20) {
      dut.clock.step(1); cycles += 1
    }
    dut.io.out.valid.expect(true.B, s"Timed out waiting for out.valid on $opName")

    println(s"--- Test: $opName ---")
    println(f"Input A : 0x$a%04X")
    println(f"Input B : 0x$b%04X")
    println(f"Expected: 0x$expected%04X")

    for (i <- 0 until dut.p.numLanes) {
      val actualVal = dut.io.out.bits.vectorOutputData(i).peek().litValue.toInt
      if (i == 0) println(f"Output  : 0x$actualVal%04X")
      dut.io.out.bits.vectorOutputData(i).expect(expected.U, s"Mismatch on lane $i")
    }
    println("---------------------\n")

    dut.clock.step(1)
  }

  // -------------------------------------------------------------------------
  // ADD tests
  // -------------------------------------------------------------------------
  it should "ADD: 1.0 + 2.0 = 3.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.add, BF16_1, BF16_2, BF16_3, "1.0 + 2.0")
    }
  }

  it should "ADD: 1.5 + 1.0 = 2.5 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.add, BF16_1_5, BF16_1, BF16_2_5, "1.5 + 1.0")
    }
  }

  it should "ADD: -0.5 + 1.0 = 0.5 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.add, BF16_N0_5, BF16_1, BF16_0_5, "-0.5 + 1.0")
    }
  }

  it should "ADD: -0.0 + 0.0 = 0.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.add, BF16_N0, BF16_0, BF16_0, "-0.0 + 0.0")
    }
  }

  // Rounding-ish boundary (still exact here): (1+eps) + eps = 1 + 2*eps
  ignore should "ADD: (1+eps) + eps = 1.0234375 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.add, BF16_1_PLUS_EPS, BF16_EPS, 0x3F83, "(1+eps) + eps")
    }
  }

  // -------------------------------------------------------------------------
  // SUB tests
  // -------------------------------------------------------------------------
  it should "SUB: 3.0 - 2.0 = 1.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.sub, BF16_3, BF16_2, BF16_1, "3.0 - 2.0")
    }
  }

  it should "SUB: 1.0 - 2.0 = -1.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.sub, BF16_1, BF16_2, BF16_N1, "1.0 - 2.0")
    }
  }

  it should "SUB: 0.0 - 0.0 = 0.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.sub, BF16_0, BF16_0, BF16_0, "0.0 - 0.0")
    }
  }

  it should "SUB: -0.0 - 0.0 = -0.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.sub, BF16_N0, BF16_0, BF16_N0, "-0.0 - 0.0")
    }
  }

  it should "SUB: 2.0 - 1.5 = 0.5 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      runOpAndCheck(dut, VPUOp.sub, BF16_2, BF16_1_5, BF16_0_5, "2.0 - 1.5")
    }
  }
}
