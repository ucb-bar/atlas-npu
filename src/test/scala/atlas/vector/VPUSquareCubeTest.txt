package atlas.vector

import chisel3._
import chisel3.simulator.EphemeralSimulator._
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers
import atlas.common.VPUParams

class VPUSquareCubeTest extends AnyFlatSpec with Matchers {

  // straight from Hao's test .... so less raw constants hopefuly
  def fpTobf(f: Float): Int = {
    val x = java.lang.Float.floatToRawIntBits(f)
    val lsb = (x >>> 16) & 1
    val roundingBias = 0x7FFF + lsb
    ((x + roundingBias) >>> 16) & 0xFFFF
  }

  def bfTofp(b: Int): Float =
    java.lang.Float.intBitsToFloat((b & 0xFFFF) << 16)
  
  def goldenSquareBF16(aBf16: Int): Int = {
    val a = bfTofp(aBf16)
    fpTobf(a * a)
  }

  def goldenCubeBF16(aBf16: Int): Int = {
    val a = bfTofp(aBf16)
    fpTobf(a * a * a)
  }

  //--- BF16 HEX Numbers ---
  // Integers
  val BF16_N2 = 0xC000 // -2.0
  val BF16_N1 = 0xBF80 // -1.0
  val BF16_N0 = 0x8000 // -0.0
  val BF16_0  = 0x0000 // +0.0
  val BF16_1  = 0x3F80 //  1.0
  val BF16_2  = 0x4000 //  2.0
  val BF16_3  = 0x4040 //  3.0
  val BF16_4  = 0x4080 //  4.0

  // Fractional numbers
  val BF16_0_5 = 0x3F00 // 0.5
  val BF16_1_5 = 0x3FC0 // 1.5
  val BF16_N1_5 = 0xBFC0 // -1.5

  val p = VPUParams()
  behavior of s"VectorEngine (BF16) square/cube lanes=${p.numLanes}"

  def initializeInterface(dut: VectorEngine, drainCycles: Int = 4): Unit = {
    dut.io.in.valid.poke(false.B)
    dut.io.out.ready.poke(true.B)
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(0.U)
      dut.io.in.bits.vecDataB(i).poke(0.U)
    }
    if (drainCycles > 0) dut.clock.step(drainCycles)
  }

  // from divsqrttest -> runs through all the lanes one by one and checks 
   def runOpAndCheckAllLanes(dut: VectorEngine, op: VPUOp.Type, a: Int, b: Int, expected: Int, opName: String): Unit = {
    // 1. Setup and Fire Input
    dut.io.in.valid.poke(true.B)
    dut.io.in.bits.instType.poke(op)
    for (i <- 0 until dut.p.numLanes) {
      dut.io.in.bits.vecDataA(i).poke(a.U)
      dut.io.in.bits.vecDataB(i).poke(b.U)
    }

    // Wait for in.ready to be true
    var cycles = 0
    while (!dut.io.in.ready.peek().litToBoolean && cycles < 20) {
      dut.clock.step(1)
      cycles += 1
    }
    dut.io.in.ready.expect(true.B, s"Timed out waiting for in.ready on $opName")
    
    dut.clock.step(1) // Advance clock to complete the input handshake (FSM goes to BUSY)
    dut.io.in.valid.poke(false.B) // Drop valid so we don't double-fire

    // 2. Wait for Output Valid (FSM finishes calculating)
    cycles = 0
    while (!dut.io.out.valid.peek().litToBoolean && cycles < 100) {
      dut.clock.step(1)
      cycles += 1
    }
    dut.io.out.valid.expect(true.B, s"Timed out waiting for out.valid on $opName")

    // 3. Print and Check Values
    println(s"--- Test: $opName ---")
    println(f"Input A : 0x$a%04X")
    println(f"Input B : 0x$b%04X")
    println(f"Expected: 0x$expected%04X")

    for (i <- 0 until dut.p.numLanes) {
      val actualVal = dut.io.out.bits.vectorOutputData(i).peek().litValue.toInt
      if (i == 0) {
        // Only print lane 0 to keep the console clean
        println(f"Output  : 0x$actualVal%04X") 
      }
      dut.io.out.bits.vectorOutputData(i).expect(expected.U, s"Mismatch on lane $i")
    }
    println("---------------------\n")

    // 4. Complete Output Handshake (FSM returns to IDLE)
    // out.ready is already poked true in initializeInterface
    dut.clock.step(1) 
  }

  // Square Tests
  it should "SQUARE: 2.0^2 = 4.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenSquareBF16(BF16_2)
      runOpAndCheckAllLanes(dut, VPUOp.square, BF16_2, 0, exp, "square(2.0)")
    }
  }

  it should "SQUARE: (-2.0)^2 = 4.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenSquareBF16(BF16_N2)
      runOpAndCheckAllLanes(dut, VPUOp.square, BF16_N2, 0, exp, "square(-2.0)")
    }
  }

  it should "SQUARE: (-0.0)^2 = +0.0 (all lanes) [sign behavior check]" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenSquareBF16(BF16_N0)
      // For IEEE, (-0)^2 should be +0. BF16 rounding should preserve +0.
      runOpAndCheckAllLanes(dut, VPUOp.square, BF16_N0, 0, exp, "square(-0.0)")
    }
  }

  it should "SQUARE: 0.5^2 = 0.25 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenSquareBF16(BF16_0_5)
      runOpAndCheckAllLanes(dut, VPUOp.square, BF16_0_5, 0, exp, "square(0.5)")
    }
  }

  // Cube -> Note this takes an addtional cycle comapred to square
  it should "CUBE: 2.0^3 = 8.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenCubeBF16(BF16_2) // should become BF16 encoding of 8.0 = 0x4100
      runOpAndCheckAllLanes(dut, VPUOp.cube, BF16_2, 0, exp, "cube(2.0)")
    }
  }

  it should "CUBE: (-2.0)^3 = -8.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenCubeBF16(BF16_N2) // should be BF16 of -8.0 = 0xC100
      runOpAndCheckAllLanes(dut, VPUOp.cube, BF16_N2, 0, exp, "cube(-2.0)")
    }
  }

  it should "CUBE: 0.0^3 = 0.0 (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenCubeBF16(BF16_0)
      runOpAndCheckAllLanes(dut, VPUOp.cube, BF16_0, 0, exp, "cube(0.0)")
    }
  }

  it should "CUBE: 1.5^3 (rounding) (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenCubeBF16(BF16_1_5)
      runOpAndCheckAllLanes(dut, VPUOp.cube, BF16_1_5, 0, exp, "cube(1.5)")
    }
  }

  it should "CUBE: (-1.5)^3 (rounding + sign) (all lanes)" in {
    simulate(new VectorEngine(p)) { dut =>
      initializeInterface(dut)
      val exp = goldenCubeBF16(BF16_N1_5)
      runOpAndCheckAllLanes(dut, VPUOp.cube, BF16_N1_5, 0, exp, "cube(-1.5)")
    }
  }
}
